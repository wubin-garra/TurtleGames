// 高性能碰撞检测 — O(n log n) 扫描线算法
function detectCollisions(turtles) {
  const sorted = [...turtles].sort((a, b) =>
    a.x - a.radius - (b.x - b.radius));

  const pairs = [];
  for (let i = 0; i < sorted.length; i++) {
    for (let j = i + 1; j < sorted.length; j++) {
      if (sorted[j].x - sorted[j].radius >
          sorted[i].x + sorted[i].radius) break;

      const dist = distance(sorted[i], sorted[j]);
      const minDist = sorted[i].radius + sorted[j].radius;
      if (dist < minDist)
        pairs.push([sorted[i], sorted[j]]);
    }
  }
  return pairs;
}
